| **üß† –≠—Ñ—Ñ–µ–∫—Ç–∏–≤–Ω–æ–µ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ –û–ó–£** | –¢–∞–∫ –∫–∞–∫ –¥–∞—Ç–∞—Å–µ—Ç –Ω–µ –∑–∞–≥—Ä—É–∂–∞–µ—Ç—Å—è –≤ –ø–∞–º—è—Ç—å, –º—ã –º–æ–∂–µ–º –æ–±—Ä–∞–±–∞—Ç—ã–≤–∞—Ç—å —Å–æ—Ç–Ω–∏ –º–∏–ª–ª–∏–æ–Ω–æ–≤ —Å—Ç—Ä–æ–∫ –Ω–∞ –º–∞—à–∏–Ω–∞—Ö —Å —É–º–µ—Ä–µ–Ω–Ω—ã–º –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ–º –û–ó–£ (–Ω–∞–ø—Ä–∏–º–µ—Ä, 32-64 –ì–ë). | **–°–Ω–∏–∂–µ–Ω–∏–µ —Ç—Ä–µ–±–æ–≤–∞–Ω–∏–π –∫ —Å–µ—Ä–≤–µ—Ä–∞–º.** –ù–µ—Ç –Ω–µ–æ–±—Ö–æ–¥–∏–º–æ—Å—Ç–∏ –∞—Ä–µ–Ω–¥–æ–≤–∞—Ç—å –∏–ª–∏ –ø–æ–∫—É–ø–∞—Ç—å –¥–æ—Ä–æ–≥–æ—Å—Ç–æ—è—â–∏–µ —Å–µ—Ä–≤–µ—Ä—ã —Å –±–æ–ª—å—à–∏–º –æ–±—ä–µ–º–æ–º –û–ó–£. |
| **üß© –ü—Ä–æ—Å—Ç–æ—Ç–∞ –∏ —Å—Ç–∞–Ω–¥–∞—Ä—Ç –∏–Ω–¥—É—Å—Ç—Ä–∏–∏** | –ë–∏–±–ª–∏–æ—Ç–µ–∫–∞ `datasets` –ø—Ä–µ–¥–æ—Å—Ç–∞–≤–ª—è–µ—Ç –ø—Ä–æ—Å—Ç–æ–π API –¥–ª—è –∑–∞–≥—Ä—É–∑–∫–∏, –ø–µ—Ä–µ–º–µ—à–∏–≤–∞–Ω–∏—è (`shuffle`), —Ä–∞–∑–¥–µ–ª–µ–Ω–∏—è (`split`) –∏ –æ–±—Ä–∞–±–æ—Ç–∫–∏ (`map`) –¥–∞–Ω–Ω—ã—Ö, –ø–æ–ª–Ω–æ—Å—Ç—å—é —Å–æ–≤–º–µ—Å—Ç–∏–º—ã–π —Å PyTorch –∏ TensorFlow. | **–£—Å–∫–æ—Ä–µ–Ω–∏–µ —Ä–∞–∑—Ä–∞–±–æ—Ç–∫–∏.** –ú—ã –∏—Å–ø–æ–ª—å–∑—É–µ–º –≥–æ—Ç–æ–≤–æ–µ, –ø—Ä–æ–≤–µ—Ä–µ–Ω–Ω–æ–µ –∏ –æ–ø—Ç–∏–º–∏–∑–∏—Ä–æ–≤–∞–Ω–Ω–æ–µ —Ä–µ—à–µ–Ω–∏–µ, –∞ –Ω–µ –ø–∏—à–µ–º —Å–æ–±—Å—Ç–≤–µ–Ω–Ω—ã–µ "–≤–µ–ª–æ—Å–∏–ø–µ–¥—ã" –¥–ª—è –∑–∞–≥—Ä—É–∑–∫–∏ –¥–∞–Ω–Ω—ã—Ö. |

### 4. –°—Ä–∞–≤–Ω–µ–Ω–∏–µ —Å –Ω–∞–∏–≤–Ω—ã–º –ø–æ–¥—Ö–æ–¥–æ–º (CSV)

### –¢–µ—Å—Ç –¥–ª—è –¥–µ–º–æ–Ω—Å—Ç—Ä–∞—Ü–∏–∏

–≠—Ç–æ—Ç Python-—Å–∫—Ä–∏–ø—Ç –Ω–∞–≥–ª—è–¥–Ω–æ –ø–æ–∫–∞–∑—ã–≤–∞–µ—Ç —Ä–∞–∑–Ω–∏—Ü—É –≤ –ø—Ä–æ–∏–∑–≤–æ–¥–∏—Ç–µ–ª—å–Ω–æ—Å—Ç–∏. –û–Ω —Å–æ–∑–¥–∞–µ—Ç —Ñ–µ–π–∫–æ–≤—ã–π –¥–∞—Ç–∞—Å–µ—Ç –Ω–∞ 1 –º–ª–Ω —Å—Ç—Ä–æ–∫, —Å–æ—Ö—Ä–∞–Ω—è–µ—Ç –µ–≥–æ –≤ CSV –∏ Parquet, –∞ –∑–∞—Ç–µ–º –∑–∞–º–µ—Ä—è–µ—Ç –≤—Ä–µ–º—è –∑–∞–≥—Ä—É–∑–∫–∏ –∏ —Ä–∞–∑–º–µ—Ä —Ñ–∞–π–ª–æ–≤.

**–ö–∞–∫ –∑–∞–ø—É—Å—Ç–∏—Ç—å:**
1.  –£—Å—Ç–∞–Ω–æ–≤–∏—Ç–µ –Ω–µ–æ–±—Ö–æ–¥–∏–º—ã–µ –±–∏–±–ª–∏–æ—Ç–µ–∫–∏:
    ```bash

    ```
2.  –°–æ—Ö—Ä–∞–Ω–∏—Ç–µ –∏ –∑–∞–ø—É—Å—Ç–∏—Ç–µ –∫–æ–¥ –Ω–∏–∂–µ.

```python
import pandas as pd

from faker import Faker # –î–ª—è –≥–µ–Ω–µ—Ä–∞—Ü–∏–∏ —Å–ª—É—á–∞–π–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö

# --- 1. –ù–∞—Å—Ç—Ä–æ–π–∫–∞ –∏ —Å–æ–∑–¥–∞–Ω–∏–µ "–±–æ–ª—å—à–æ–≥–æ" –∏–≥—Ä—É—à–µ—á–Ω–æ–≥–æ –¥–∞—Ç–∞—Å–µ—Ç–∞ ---

NUM_ROWS = 1_000_000

print(f"–°–æ–∑–¥–∞–µ–º –∏–≥—Ä—É—à–µ—á–Ω—ã–π –¥–∞—Ç–∞—Å–µ—Ç –Ω–∞ {NUM_ROWS:,} —Å—Ç—Ä–æ–∫...")

# –ü—Ä–æ—Å—Ç—ã–µ SMILES –¥–ª—è –ø—Ä–∏–º–µ—Ä–∞

    'CCO', 'c1ccccc1', 'CN1C=NC2=C1C(=O)N(C)C(=O)N2C',
    'CC(=O)OC1=CC=CC=C1C(=O)O', 'NC(C)C(=O)O'
]

# –°–æ–∑–¥–∞–µ–º DataFrame
df = pd.DataFrame({
    # –ü–æ–≤—Ç–æ—Ä—è–µ–º –ø—Ä–æ—Å—Ç—ã–µ SMILES, —á—Ç–æ–±—ã —Å–∏–º—É–ª–∏—Ä–æ–≤–∞—Ç—å —Ä–µ–∞–ª—å–Ω—ã–µ –¥–∞–Ω–Ω—ã–µ
    'smiles': np.random.choice(sample_smiles, NUM_ROWS),
    # –î–æ–±–∞–≤–∏–º –µ—â–µ –æ–¥–Ω—É –∫–æ–ª–æ–Ω–∫—É, —á—Ç–æ–±—ã –ø–æ–∫–∞–∑–∞—Ç—å –ø—Ä–µ–∏–º—É—â–µ—Å—Ç–≤–æ –∫–æ–ª–æ–Ω–æ—á–Ω–æ–≥–æ —Ñ–æ—Ä–º–∞—Ç–∞
    'property_value': np.random.rand(NUM_ROWS)

print("-" * 40)

print("--- –î–µ–º–æ–Ω—Å—Ç—Ä–∞—Ü–∏—è: –ù–∞–∏–≤–Ω—ã–π –ø–æ–¥—Ö–æ–¥ (–∑–∞–≥—Ä—É–∑–∫–∞ CSV –≤ Pandas) ---")
try:
    start_time = time.time()
    # –ü–æ–ø—ã—Ç–∫–∞ –ø—Ä–æ—á–∏—Ç–∞—Ç—å –≤–µ—Å—å —Ñ–∞–π–ª –≤ –ø–∞–º—è—Ç—å
    df_from_csv = pd.read_csv(CSV_PATH)
    load_csv_time = time.time() - start_time
    print(f"‚úÖ CSV –∑–∞–≥—Ä—É–∂–µ–Ω –≤ Pandas –∑–∞: {load_csv_time:.4f} —Å–µ–∫.")

    print("‚ùå –û—à–∏–±–∫–∞: –ù–µ —Ö–≤–∞—Ç–∏–ª–æ –ø–∞–º—è—Ç–∏ –¥–ª—è –∑–∞–≥—Ä—É–∑–∫–∏ CSV —Ñ–∞–π–ª–∞!")
print()


print("--- –î–µ–º–æ–Ω—Å—Ç—Ä–∞—Ü–∏—è: –†–µ–∫–æ–º–µ–Ω–¥—É–µ–º—ã–π –ø–æ–¥—Ö–æ–¥ (Parquet + `datasets`) ---")
start_time = time.time()
# –ó–∞–≥—Ä—É–∑–∫–∞ —Å –ø–æ–º–æ—â—å—é memory-mapping. –ü—Ä–æ–∏—Å—Ö–æ–¥–∏—Ç –º–≥–Ω–æ–≤–µ–Ω–Ω–æ.

print(f"   –ü—Ä–∏ —ç—Ç–æ–º –û–ó–£ –ø—Ä–∞–∫—Ç–∏—á–µ—Å–∫–∏ –Ω–µ –∏—Å–ø–æ–ª—å–∑—É–µ—Ç—Å—è.")
print("\n–û–±—ä–µ–∫—Ç –¥–∞—Ç–∞—Å–µ—Ç–∞ –≥–æ—Ç–æ–≤ –∫ —Ä–∞–±–æ—Ç–µ:")

print("-" * 40)
print("–í—Å–µ —Ñ–∞–π–ª—ã —Å–æ–∑–¥–∞–Ω—ã. –ú–æ–∂–µ—Ç–µ —É–¥–∞–ª–∏—Ç—å –∏—Ö:")
print(f"rm {CSV_PATH} {PARQUET_PATH}")
```
